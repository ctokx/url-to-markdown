terminal Terminal 
$ npm install -g md4llm

$ md4llm https://docs.python.org -s "#content"

# Convert with metadata as JSON
$ md4llm https://example.com --meta --format json

# Batch process URLs
$ md4llm --batch urls.txt -o ./output/
